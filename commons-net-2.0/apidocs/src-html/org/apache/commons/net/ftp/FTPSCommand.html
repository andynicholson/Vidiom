018    package org.apache.commons.net.ftp;
019    
020    /**
021     * I acquire a command added in FTPS.
022     * @since 2.0
023     */
024    public final class FTPSCommand {
025        public static final int AUTH = 0;
026        public static final int ADAT = 1;
027        public static final int PBSZ = 2;
028        public static final int PROT = 3;
029        public static final int CCC = 4;
030    
031        public static final int AUTHENTICATION_SECURITY_MECHANISM = AUTH;
032        public static final int AUTHENTICATION_SECURITY_DATA = ADAT;
033        public static final int PROTECTION_BUFFER_SIZE = PBSZ;
034        public static final int DATA_CHANNEL_PROTECTION_LEVEL = PROT;
035        public static final int CLEAR_COMMAND_CHANNEL = CCC;
036    
037        static final String[] _commands = {"AUTH","ADAT","PBSZ","PROT","CCC"};
038    
039        /**
040         * Retrieve the FTPS command string corresponding to a specified
041         * command code.
042         * <p>
043         * @param command The command code.
044         * @return The FTPS command string corresponding to a specified 
045         *  command code.
046         */
047        public static final String getCommand(int command) {
048            return _commands[command];
049        }
050    }
